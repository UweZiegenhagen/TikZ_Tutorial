	\draw[black] (o) circle (\distance mm);
}

\readlist\directions{right;above;left;below}
\foreachitem\direction\in\directions{
  \foreachitem\distance\in\distances{
    \node[\direction=\distance cm of o] {\distancecnt};
   }}
\end{lstlisting}
